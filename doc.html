Explanation of the system of categories in which are main class methods

Methods are categorised by their function. These categories are "add", "get", "do".

Function which aren't in any category

async function:

description:

This function can make your code asynchronus

important parameters:

  1. time out - the number of cycles to be performed before the time out function is called
  2. interval - the number of cycles per second
  3. time out function - the function called when number of cycles is bigger then a time out

optional:

  1. inifinite amount - the functions (When the function return true, cycle will stop)

example:


	var a = false;
	button.addEventLisenter("click", function(){
		a = true;
	});
	async(1000, 33, function(){// This function is called when a timeout occurs (cycle stops).
		alert("time out");
	}, function(){ // This function is called every cycle (in this case 33/s);
		if(a === true){ 
			alert("hello world");
			return true; // This will end cycle
		}
	});

function async ( /**/ ) {
    var interval;
    var time;
    var timeOutCallBack;
    var args = arguments;
    if (typeof args[0] == "number" && typeof args[1] == "number" && typeof args[2] == "function") {
        time = args[0];
        interval = args[1]
        timeOutCallBack = args[2]
        var timeWatchdog = 0;
        var interval = setInterval(function () {
            for (var i = 0; i < args.length; i++) {
                if (i != 0 && i != 1 && i != 2) {
                    if (typeof args[i] == "function") {
                        if (args[i](timeWatchdog) == true) {
                            clearInterval(interval);
                        }
                    }
                }
            }

            if (time <= timeWatchdog) {
                clearInterval(interval);
                timeOutCallBack();
            } else {
                timeWatchdog = timeWatchdog + 1;
            }
        }, interval)
    } else {
        console.error("wrong arguments")
    }
}
